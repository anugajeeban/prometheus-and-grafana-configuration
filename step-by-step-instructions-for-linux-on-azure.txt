*Create Prometheus service
vim /etc/systemd/system/prometheus.service

*DEFINE SYSTEM UNIT FILE TO CONTROL THE DAEMON USING SYSTEMD

[Unit]
Description=Prometheus
Wants=network-online.target
After=network-online.target

[Service]
User=prometheus
Group=prometheus
Type=simple
ExecStart=/usr/local/bin/prometheus \
    --config.file /etc/prometheus/prometheus.yml \
    --storage.tsdb.path /var/lib/prometheus/ \
    --web.console.templates=/etc/prometheus/consoles \
    --web.console.libraries=/etc/prometheus/console_libraries

[Install]
WantedBy=multi-user.target

*RELOAD SYSTEMD DAEMON AND START SERVICE

sudo systemctl daemon-reload
sudo systemctl enable prometheus
sudo systemctl start prometheus
sudo systemctl status prometheus
sudo journalctl -u prometheus

*//Step 5: Install Blackbox Exporter https://blog.ruanbekker.com/blog/2019/05/17/install-blackbox-exporter-to-monitor-websites-with-prometheus/


*//download and unzip the blackbox_exporter-0.19.0.linux-amd64.tar.gz

*//run the blackbox_exporter Note: add & to run the script in the background
./blackbox_exporter&
*//edit prometheus.yml to add the exporter module https://github.com/anugajeeban/blackbox_exporter
- job_name: 'blackbox'
    metrics_path: /probe
    params:
      module: [http_2xx]  # Look for a HTTP 200 response.
    static_configs:
      - targets:
        - http://prometheus.io    # Target to probe with http.
        - https://prometheus.io   # Target to probe with https.
        - http://www.google.com 
        - http://youtube.com
        - https://www.canadiantire.ca/

    relabel_configs:
      - source_labels: [__address__]
        target_label: __param_target
      - source_labels: [__param_target]
        target_label: instance
      - target_label: __address__
        replacement: 127.0.0.1:9115  # The blackbox exporter's real hostname:port.
